```


svc=crash-myad

stop:
	taskid=$$(screen -ls | awk '{print $$1}' | grep "$(svc)")
	if [ ! "$${taskid}" == "" ];
	then
		screen -x $${taskid} -X quit
	fi
       
start:
	screen -dmS $(svc) bash -c "python3 app.py"
    
restart: stop start

.ONESHELL:
```
